CREATE SCHEMA JOGO;

USE JOGO;

CREATE TABLE PERSONAGEM (
  PERSONAGEM_ID BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  NOME_PERSONAGEM VARCHAR(20) NOT NULL,
  CLASSE_PERSONAGEM VARCHAR(20),
  CONSTRAINT PERSONAGEMID_PK PRIMARY KEY (PERSONAGEM_ID)
);

CREATE TABLE ITEM (
  ITEM_ID BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  ITEM_NOME VARCHAR(20) NOT NULL,
  ITEM_TIPO VARCHAR(20),
  CONSTRAINT ITEM_PK PRIMARY KEY (ITEM_ID)
);

CREATE TABLE INVENTARIO (
  PERSONAGEM_ID BIGINT UNSIGNED NOT NULL,
  ITEM_ID BIGINT UNSIGNED NOT NULL,
  PRIMARY KEY (PERSONAGEM_ID, ITEM_ID),
  FOREIGN KEY PERSONAGEM_FK (PERSONAGEM_ID) REFERENCES PERSONAGEM (PERSONAGEM_ID) ON DELETE RESTRICT ON UPDATE CASCADE,
  FOREIGN KEY ITEM_FK (ITEM_ID) REFERENCES ITEM (ITEM_ID) ON DELETE RESTRICT ON UPDATE CASCADE
);
